\def\@makechapterhead#1{%
  \vspace*{\ChapterBeforeSkip}%
  {\parindent\z@\centering\normalfont\bfseries\color{chaptercolor}%
    \ifnum\c@secnumdepth>\m@ne
      \Large Chapter -- \thechapter\par
      \vskip \ChapterNumTitleGap
    \fi
    \Huge #1\par
    \vskip \ChapterTitleRuleGap
    {\color{chaptercolor}\hrule height 2.5pt\relax}%
    \par\nobreak
    \vskip \ChapterAfterSkip
  }%
}
